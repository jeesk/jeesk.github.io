```
public class TransformDemo {
	@Test
	public void testTransf() {
		// 1; 获得连接
		Connection conn = DbUtil.getConnection();
		// 查询张三是否满足转账的条件
		//	判断转账条件否满足
		String sql = "select * from account where name = '小明' and money >=20000";
		PreparedStatement prepareStatement = null;
		try {
			prepareStatement = conn.prepareStatement(sql);
			ResultSet ret = prepareStatement.executeQuery();
			while (!ret.next()) {

				System.out.println("余额不足, 转账失败");
			}
		} catch (SQLException e1) {
			e1.printStackTrace();
		}

		try {
			//	开启事务
			conn.setAutoCommit(false);
			System.out.println("开始 转账操作");
			sql = "update account set money = 10000 where name = '小明'";
			// 执行转账操作,减少小明账户里面的钱
			prepareStatement = conn.prepareStatement(sql);
			prepareStatement.executeUpdate();
			
			int i = 1 / 0; // 如果中断了,开启了事务则会回滚数据,执行小明的操作会回滚
			
			
			//  修改仓老师账户里面的钱
			sql = "update account set money = money+10000 where name = '仓老师'";
			prepareStatement = conn.prepareStatement(sql);
			prepareStatement.executeUpdate();
			// 提交事务
			conn.commit();
			System.out.println("转账成功");
		} catch (SQLException e) {
			e.printStackTrace();
			try {
				conn.rollback();
				// 回滚事物
			} catch (SQLException e1) {
				e1.printStackTrace();
			} finally {
				// 释放资源
				DbUtil.close(prepareStatement, null, conn);
			}
		}

	}
}
```

#### 一、事务概述
事务(Transaction,简写为tx):
在数据库中,所谓事务是指一组逻辑操作,不论成功与失败都作为一个整体进行工作,要么全部执行，要么全不执行。
简单理解为，由一条或多条SQL语句组成的一组的操作(比如说转账)，在执行时，要么都成功，要么都失败。
如果成功就确认操作，称为提交commit；
失败就恢复数据/撤销，称为回滚rollback

#### 二、处理事务的两个动作:
       提交/确认:commit:    当整个事务中,所有的逻辑操作都正常执行成功.       ---->提交事务.
       回滚/撤销:rollback:   当整个事务中,有一个或多个逻辑操作执行失败,       ---->回滚事务. 撤销该事务中的所有操作,恢复到最初的状态.

####　三、事务的ACID属性:
1. 原子性（Atomicity）:事务是应用中不可再分的最小逻辑执行单位体，要么都执行，要么都不执行； 
2. 一致性（Consistency）:事务结束后，数据库内的数据是合法正确的、不被破坏的；(例如A账户1000，B账户1000元，转账前后共2000元应该保持一致)
3. 隔离性（Isolation）: 并发执行的事务之间彼此相互独立、互不干扰； (学持久层框架时再讲)
4. 持久性（Durability）:事务提交后，数据会持久化到数据库中,不可再回滚了；  

事务的并发性造成的问题、数据库的隔离级别后面学持久层框架时再讲

#### 四、如何在代码中去处理事务:
操作事务的API
1.在JDBC中缺省情况下,事务是自动提交的,控制事务必须先设置事务为手动提交.
connection对象.setAutoCommit(false);//设置事务为手动提交.
2.提交事务.
connection对象.commit();
3.若出现异常必须回滚事务:
connection对象.rollback();

那接下来就可以写代码实现了！


####　五、注意事项

1.　在JDBC在事务是默认提交的,那是在什么时候提交的?在执行一个DML/DDL操作的时候,就已经提交事务了.
2.　针对于CRUD操作,DML操作才需要事务,查询操作不需要事务(但也可以加上事务).

3.不回滚事务,发现总余额依然是正确的(因为出现异常后没有提交),但是还是要回滚事务,因为不回滚不会释放数据库资源.

4.　MySQL中,InnoDB支持外键.支持事务;MyISAM不支持外键,不支持事务.(alter table account engine = 'myisam')

5.　以后事务我们不应该在DAO层处理,应该在service层控制.
6.　以后,凡是发现自己编写的代码是正确的,测试也通过,但是就是数据库表中的数据不变----->这时要记得看看事务有没有提交?.

7.　更加详细的事务相关知识在MyBatis,Spring,项目的时候都会再讲.

在开发中,获取自动生成的主键的案例很多,比如后面要学习的单据和单据明细之间的关系.

 ```
Connection.prepareStatement(String sql,int autoGeneratedKeys) :创建PreparedStatement对象,指定是否需要返回生成的主键. 

   常量值:Statement.RETURN_GENERATED_KEYS
PreparedStatement .getGeneratedKeys():获取自动生成的主键返回ResultSet
 
 ```
MyBatis中获取自动生成主键,原理就是下面的代码
```

package com.shanjiancao.smis.test;

import static org.junit.Assert.*;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

import org.junit.Test;

import com.mysql.jdbc.Statement;
import com.shanjiancao.smis.util.DbUtil;

/**
 * GetGeneratedKeyDemo.java
 * @version
 * @date 2018年5月26日 下午1:15:16
 * @author jeesk
 * @since 1.0
 *  在执行插入操作的时候获得自动生成的主键演示\
 *  
 *  步骤:
 *  1);  创建一个名字为小红的账户, 拥有金钱为50000
 *  2):  获得连接
 *  3):  在获得sql语句对象 的时候 ,指定要生成主键
 *  4):  执行插入语句	
 *  5):  获得生成的主键的值
 *  6):  关闭资源
 */
public class GetGeneratedKeyDemo {
	@Test
	public void testGetGeneratedKey() throws Exception {
		// 创建一个名字为小红的账户, 拥有金钱为50000
		String sql = "insert into account values('小红',50000)";
		// 获得连接
		Connection conn = DbUtil.getConnection();
		// 在获得sql语句对象 的时候 ,指定要生成主键
		PreparedStatement prepareStatement = conn.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS);
		// 执行插入语句	
		prepareStatement.executeUpdate();
		// 获得生成的主键的值
		ResultSet rs = prepareStatement.getGeneratedKeys();

		while (rs.next()) {
			Long key = rs.getLong(1);
			System.out.println("生成的主键 是: " + key);

		}
		// 关闭资源
		DbUtil.close(prepareStatement, rs, conn);

	}
}
/*
 * 小结: 
 * 		如果要获得执行插入语句自动 生成的主键, 可以在获得sql语句对象 的时候指定 
 * 		connection.prepareStatement(sql,Statement.RETURN_GENERATED_KEYS);
 * 		在执行了插入语句后,用preparedStatement获得返回的主键的ResultSet,遍历获得主键的值 
 * 
 * */
```

### 使用内省获把ResultSet转换成javaBean对象

```
public interface IResultSetLHandler<T> {
	
	T hanlder(ResultSet rs);
}
```
```
package com.shanjiancao.smis.hanlde;

import java.beans.BeanInfo;
import java.beans.IntrospectionException;
import java.beans.Introspector;
import java.beans.PropertyDescriptor;
import java.lang.reflect.InvocationTargetException;
import java.sql.ResultSet;
import java.sql.SQLException;

/**
 * StudentResultSetHandlerImpl.java
 * @version
 * @date 2018年5月26日 上午10:15:57
 * @author jeesk
 * @since 1.0
 *  将传入的ResultSet转为指定的JavaBaean对象
 */
public class StudentResultSetHandlerImpl<T> implements IResultSetLHandler<T> {

	private Class<T> t;

	public StudentResultSetHandlerImpl(Class<T> t) {
		this.t = t;
	}

	@Override
	public T hanlder(ResultSet rs) {

		T newInstance = null;
		try {
			newInstance = t.newInstance();
		} catch (InstantiationException | IllegalAccessException e1) {
			e1.printStackTrace();
		}
		try {
			while (rs.next()) {
				//	获得这一对象的JavaBean
				BeanInfo beanInfo = Introspector.getBeanInfo(newInstance.getClass(), Object.class);
				PropertyDescriptor[] propertyDescriptors = beanInfo.getPropertyDescriptors();
				for (PropertyDescriptor propertyDescriptor : propertyDescriptors) {
					// 遍历属性 获得这一个属性的写方法， 使用这一个属性的name作为key从Resultset获得对应的Value，然后将这一个value设置到JavaBean对象的属性
					propertyDescriptor.getWriteMethod().invoke(newInstance, rs.getObject(propertyDescriptor.getName()));
				}

			}
		} catch (SQLException e) {
			e.printStackTrace();
		} catch (IllegalAccessException e) {
			e.printStackTrace();
		} catch (IntrospectionException e) {
			e.printStackTrace();
		} catch (IllegalArgumentException e) {
			e.printStackTrace();
		} catch (InvocationTargetException e) {
			e.printStackTrace();
		}
		return newInstance;
	}

}

/*
 * 使用内省将Result转换成JavaBean
 *  1): 获得指定对象的PropertyDescriptors(); 然后遍历
 *  2): 先获得这一个属性的name,先name作为key获得对应的value
 *  3); 获得这一个属性的读方法,将value作为参数设置给javaBean
 *  4): 每一个Resulet遍历一次可以获得一个对象, 如果获得多个对象,可以将对象添加到集合中去
 * 
 * 
 * 
 * */
```
